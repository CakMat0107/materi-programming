import sqlite3

def init_db():
    conn = sqlite3.connect('santri_app.db')
    cursor = conn.cursor()
    cursor.execute('''CREATE TABLE IF NOT EXISTS santri (nis TEXT PRIMARY KEY, nama TEXT, kelas TEXT)''')
    conn.commit()
    conn.close()
